# list 
grocery = ["milk", "eggs", "bread", "fruits", "veggies", "drinks"]
print(grocery)



# printing only odd list values
grocery = ["milk", "eggs", "bread", "fruits", "veggies", "drinks"]
for i in range (3): 
    print(grocery[2*i])
